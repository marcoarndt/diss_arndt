# .latexmkrc Konfiguration

# Sicherstellen, dass lualatex genutzt wird
$pdf_mode = 4;
$postlatex_processing_filter = 'makeglossaries';

# Abhängigkeiten für Standard-Glossare (Abkürzungen, Glossar)
add_cus_dep('glo', 'gls', 0, 'makeglossaries');
add_cus_dep('acn', 'acr', 0, 'makeglossaries');

# WICHTIG: Abhängigkeiten für deine EIGENEN Verzeichnisse (Indizes, Formeln)
add_cus_dep('indices-glo', 'indices-gls', 0, 'makeglossaries');
add_cus_dep('formulas-glo', 'formulas-gls', 0, 'makeglossaries');

# Die Funktion, die aufgerufen wird
sub makeglossaries {
    # Argument ist der Dateiname ohne Endung (Root filename)
    if ( $silent ) {
        system "makeglossaries -q \"$_[0]\"";
    }
    else {
        system "makeglossaries \"$_[0]\"";
    }
}
